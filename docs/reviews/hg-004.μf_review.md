def _weak(pattern: str, fix: str, ignore_case: bool = False) -> Tuple[re.Pattern, str]:
    """WEAK パターン生成ヘルパー"""
    flags = re.IGNORECASE if ignore_case else 0
    return (re.compile(pattern, flags), f"WHAT: {fix}")

WEAK_PURPOSE_PATTERNS = [
    _weak(r"を表す", "'を表す' → 'を可能にする' etc."),
    _weak(r"を保持する", "'を保持する' → 'を統一的に扱う' etc."),
    _weak(r"を提供する", "'を提供する' → 'を生成する' etc."),
    _weak(r"を定義する$", "'を定義する' → 'を可能にする' etc."),
    _weak(r"^データクラス$", "具体的な目的がない"),
    _weak(r"^列挙型$", "具体的な目的がない"),
    # v3.0: 英語 WEAK パターン (/dia+ レビューで追加)
    _weak(r"^Represents?\b", "'Represents' → state WHY it exists", ignore_case=True),
    _weak(r"^Holds?\b", "'Holds' → state WHY it's needed", ignore_case=True),
    _weak(r"^Provides?\b", "'Provides' → state WHY it matters", ignore_case=True),
    _weak(r"^Defines?\b", "'Defines' → state WHY it enables", ignore_case=True),
